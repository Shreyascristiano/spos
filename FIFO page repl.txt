/*
    FIFO - Elements removed on the basis of FIFO. The oldest ele is removed first when space is full
* */
public class Main {



    public static void main(String[] args) {

        Scanner z = new Scanner(System.in);
        System.out.println("HELLO mom");
        int numOfFrames, numberOfPages;
        System.out.println("Enter num of frames");
        numOfFrames = z.nextInt();
        System.out.println("Enter num of pages");
        numberOfPages = z.nextInt();
        int[] pages = new int[numberOfPages];
        System.out.println("Enter page num one by one");
        for (int i = 0; i < pages.length; i++) {
            int page = z.nextInt();
            pages[i] = page;
        }

        //frames
        LinkedList<Integer> frames = new LinkedList<>();
        int pageFault = 0;

        for (int i = 0; i < pages.length; i++) {
            int page = pages[i];
            if(!frames.contains(page)){
                if(frames.size()<numOfFrames){
                    frames.add(page);
                }else{
                    frames.removeFirst();
                    frames.addLast(page);
                }
                pageFault++;
            }
            printLinkedList(frames);
        }
        System.out.println("Page faults: "+pageFault);


    }
    private static void printLinkedList(LinkedList<Integer> frames) {
        System.out.print("Frames: ");
        for(int i : frames){
            System.out.print(i+" ");
        }
        System.out.println();
    }

}
